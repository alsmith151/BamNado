31/// Represents the strand of a genomic feature.
32#[derive(Debug, Clone, PartialEq, Eq, Copy)]
33pub enum Strand {
34    /// Forward strand (+).
35    Forward,
36    /// Reverse strand (-).
37    Reverse,
38    /// Both strands (unstranded).
39    Both,
40}
41impl FromStr for Strand {
42    type Err = String;
43
44    fn from_str(s: &str) -> Result<Self, Self::Err> {
45        match s.to_lowercase().as_str() {
46            "forward" => Ok(Strand::Forward),
47            "reverse" => Ok(Strand::Reverse),
48            "both" => Ok(Strand::Both),
49            _ => Err(format!("Invalid strand value: {s}")),
50        }
51    }
52}
53impl std::fmt::Display for Strand {
54    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
55        match self {
56            Strand::Forward => write!(f, "forward"),
57            Strand::Reverse => write!(f, "reverse"),
58            Strand::Both => write!(f, "both"),
59        }
60    }
61}

214/// Reads the header of a BAM file.
215///
216/// If the `noodles` crate fails to read the header, it falls back to `samtools`.
217/// This is useful for handling BAM files with slightly non-standard headers (e.g., from CellRanger).
218pub fn bam_header(file_path: PathBuf) -> Result<sam::Header> {
219    // Read the header of a BAM file
220    // If the noodles crate fails to read the header, we fall back to samtools
221    // This is a bit of a hack but it works for now
222    // The noodles crate is more strict about the header format than samtools
223
224    // Check that the file exists
225    if !file_path.exists() {
226        return Err(anyhow::Error::from(std::io::Error::new(
227            std::io::ErrorKind::NotFound,
228            format!("File not found: {}", file_path.display()),
229        )));
230    };
231
232    let mut reader = bam::io::indexed_reader::Builder::default()
233        .build_from_path(file_path.clone())
234        .expect("Failed to open file");
235
236    let header = match reader.read_header() {
237        std::result::Result::Ok(header) => header,
238        Err(e) => {
239            debug!("Failed to read header using noodels falling back to samtools: {e}");
240
241            let header_samtools = std::process::Command::new("samtools")
242                .arg("view")
243                .arg("-H")
244                .arg(file_path.clone())
245                .output()
246                .expect("Failed to run samtools")
247                .stdout;
248
249            let header_str =
250                String::from_utf8(header_samtools).expect("Failed to convert header to string");
251
252            // Slight hack here for CellRanger BAM files that are missing the version info
253            let header_string =
254                header_str.replace("@HD\tSO:coordinate\n", "@HD\tVN:1.6\tSO:coordinate\n");
255            let header_str = header_string.as_bytes();
256            let mut reader = sam::io::Reader::new(header_str);
257            reader
258                .read_header()
259                .expect("Failed to read header with samtools")
260        }
261    };
262    Ok(header)
263}

265/// Stores statistics and metadata about a BAM file.
266pub struct BamStats {
267    // BAM file stats
268
269    // File path
270    #[allow(dead_code)]
271    file_path: PathBuf,
272
273    // Header
274    header: sam::Header,
275
276    // Contigs present in the BAM file
277    #[allow(dead_code)]
278    contigs: Vec<Contig<String, ReqStrand>>,
279
280    // Chromosome stats
281    chrom_stats: HashMap<String, ChromosomeStats>,
282
283    // Total number of reads
284    n_reads: u64,
285    // Number of reads mapped to the genome
286    n_mapped: u64,
287    // Number of reads unmapped to the genome
288    n_unmapped: u64,
289}

291impl BamStats {
292    /// Creates a new `BamStats` instance from a BAM file path.
293    ///
294    /// This reads the header and index to calculate statistics.
295    pub fn new(file_path: PathBuf) -> Result<Self> {
296        // Read the header of the BAM file
297        let header = bam_header(file_path.clone())?;
298
299        // Get the contigs from the header
300        let contigs = header
301            .reference_sequences()
302            .iter()
303            .map(|(name, map)| {
304                let name = name.to_string();
305                let length = map.length().get();
306                Contig::new(name, 1, length, ReqStrand::Forward)
307            })
308            .collect();
309
310        // Get the index of the BAM file
311        let bam_reader = bam::io::indexed_reader::Builder::default()
312            .build_from_path(file_path.clone())
313            .expect("Failed to open file");
314        let index = bam_reader.index();
315
316        // Get the chromosome stats
317        let mut chrom_stats = HashMap::default();
318
319        for ((reference_sequence_name_buf, reference_sequence), index_reference_sequence) in header
320            .reference_sequences()
321            .iter()
322            .zip(index.reference_sequences())
323        {
324            let reference_sequence_name = std::str::from_utf8(reference_sequence_name_buf)
325                .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidInput, e))?;
326
327            let (mapped_record_count, unmapped_record_count) = index_reference_sequence
328                .metadata()
329                .map(|m| (m.mapped_record_count(), m.unmapped_record_count()))
330                .unwrap_or_default();
331
332            let stats = ChromosomeStats {
333                chrom: reference_sequence_name.to_string(),
334                length: usize::from(reference_sequence.length()) as u64,
335                mapped: mapped_record_count,
336                unmapped: unmapped_record_count,
337            };
338
339            chrom_stats.insert(reference_sequence_name.to_string(), stats);
340        }
341
342        let mut unmapped_record_count = index.unplaced_unmapped_record_count().unwrap_or_default();
343
344        let mut mapped_record_count = 0;
345
346        for (_, stats) in chrom_stats.iter() {
347            mapped_record_count += stats.mapped;
348            unmapped_record_count += stats.unmapped;
349        }
350
351        let n_reads = mapped_record_count + unmapped_record_count;
352
353        Ok(Self {
354            file_path,
355            header,
356            contigs,
357            chrom_stats,
358            n_reads,
359            n_mapped: mapped_record_count,
360            n_unmapped: unmapped_record_count,
361        })
362    }

364    /// Estimates the optimal genome chunk length for parallel processing.
365    ///
366    /// The chunk length is calculated based on the genome size, number of mapped reads,
367    /// and the desired bin size.
368    pub fn estimate_genome_chunk_length(&self, bin_size: u64) -> Result<u64> {
369        // genomeLength = sum(bamHandles[0].lengths)
370        // max_reads_per_bp = max([float(x) / genomeLength for x in mappedList])
371        // genomeChunkLength = int(min(5e6, int(2e6 / (max_reads_per_bp * len(bamHandles)))))
372        // genomeChunkLength -= genomeChunkLength % tile_size
373
374        let stats = &self.chrom_stats;
375        let genome_length = stats.values().map(|x| x.length).sum::<u64>();
376        let max_reads_per_bp = self.n_mapped as f64 / genome_length as f64;
377        let genome_chunk_length = 5e6_f64.min(2e6 / (max_reads_per_bp));
378
379        let correction = genome_chunk_length % bin_size as f64;
380        let genome_chunk_length = genome_chunk_length - correction;
381
382        Ok(genome_chunk_length as u64)
383    }

385    /// Splits the genome into chunks for parallel processing.
386    pub fn genome_chunks(&self, bin_size: u64) -> Result<Vec<Region>> {
387        let genome_chunk_length = self.estimate_genome_chunk_length(bin_size)?;
388        let chrom_chunks = self
389            .chrom_stats
390            .iter()
391            .flat_map(|(chrom, stats)| {
392                let mut chunks = Vec::new();
393                let mut chunk_start = 1;
394                let chrom_end = stats.length;
395
396                while chunk_start <= chrom_end {
397                    // Corrected to include the last position in the range
398                    let chunk_end = chunk_start + genome_chunk_length - 1; // Adjust to ensure the chunk covers exactly genome_chunk_length positions
399                    let chunk_end = chunk_end.min(chrom_end); // Ensure we do not exceed the chromosome length
400
401                    let start = Position::try_from(chunk_start as usize).unwrap();
402                    let end = Position::try_from(chunk_end as usize).unwrap();
403
404                    let region = Region::new(&*chrom.clone(), start..=end);
405                    chunks.push(region);
406                    chunk_start = chunk_end + 1; // Corrected to start the next chunk right after the current chunk ends
407                }
408                chunks
409            })
410            .collect();
411
412        Ok(chrom_chunks)
413    }

561/// Converts a list of regions to a `Lapper` interval tree for efficient overlap queries.
562pub fn regions_to_lapper(regions: Vec<Region>) -> Result<HashMap<String, Lapper<usize, u32>>> {
563    let mut lapper: HashMap<String, Lapper<usize, u32>> = HashMap::default();
564    let mut intervals: HashMap<String, Vec<Iv>> = HashMap::default();
565
566    for reg in regions {
567        let chrom = reg.name().to_string();
568        let start = reg.start().map(|x| x.get());
569        let end = reg.end().map(|x| x.get());
570
571        let start = match start {
572            Bound::Included(start) => start,
573            Bound::Excluded(start) => start + 1,
574            _ => 0,
575        };
576
577        let end = match end {
578            Bound::Included(end) => end,
579            Bound::Excluded(end) => end - 1,
580            _ => 0,
581        };
582
583        let iv = Iv {
584            start,
585            stop: end,
586            val: 0,
587        };
588        intervals.entry(chrom.clone()).or_default().push(iv);
589    }
590
591    for (chrom, ivs) in intervals.iter() {
592        let lap = Lapper::new(ivs.to_vec());
593        lapper.insert(chrom.clone(), lap);
594    }
595
596    Ok(lapper)
597}

599/// Reads a BED file and converts it to a `Lapper` interval tree.
600pub fn bed_to_lapper(bed: PathBuf) -> Result<HashMap<String, Lapper<usize, u32>>> {
601    // Read the bed file and convert it to a lapper
602    let reader = std::fs::File::open(bed)?;
603    let buf_reader = std::io::BufReader::new(reader);
604    let mut bed_reader = bed::io::Reader::<4, _>::new(buf_reader);
605    let mut record = bed::Record::default();
606    let mut intervals: HashMap<String, Vec<Iv>> = HashMap::default();
607    let mut lapper: HashMap<String, Lapper<usize, u32>> = HashMap::default();
608
609    while bed_reader.read_record(&mut record)? != 0 {
610        let chrom = record.reference_sequence_name().to_string();
611        let start = record.feature_start()?;
612        let end = record
613            .feature_end()
614            .context("Failed to get feature end")??;
615
616        let iv = Iv {
617            start: start.get(),
618            stop: end.get(),
619            val: 0,
620        };
621        intervals.entry(chrom.clone()).or_default().push(iv);
622    }
623
624    for (chrom, ivs) in intervals.iter() {
625        let lap = Lapper::new(ivs.to_vec());
626        lapper.insert(chrom.clone(), lap);
627    }
628    // Check if the lapper is empty
629    if lapper.is_empty() {
630        return Err(anyhow::Error::msg("Lapper is empty"));
631    }
632
633    Ok(lapper)
634}

636/// Converts the chromosome names in a `Lapper` to chromosome IDs based on a BAM file's header.
637///
638/// This is useful for when we want to use the lapper with the BAM file.
639pub fn convert_lapper_chrom_names_to_ids(
640    lapper: HashMap<String, Lapper<usize, u32>>,
641    bam_stats: &BamStats,
642) -> Result<HashMap<usize, Lapper<usize, u32>>> {
643    // Convert the chromosome names in the lapper to the chromosome IDs in the BAM file
644    let chrom_id_mapping = bam_stats.chromosome_name_to_id_mapping()?;
645    let mut lapper_chrom_id: HashMap<usize, Lapper<usize, u32>> = HashMap::default();
646    for (chrom, lap) in lapper.iter() {
647        let chrom_id = chrom_id_mapping.get(chrom).ok_or_else(|| {
648            anyhow::Error::msg(format!("Chromosome {chrom} not found in BAM file"))
649        })?;
650        lapper_chrom_id.insert(*chrom_id, lap.clone());
651    }
652    Ok(lapper_chrom_id)
653}

655/// Reads the header of a BAM file.
656///
657/// If the `noodles` crate fails to read the header, it falls back to `samtools`.
658/// This is useful for handling BAM files with slightly non-standard headers (e.g., from CellRanger).
659pub fn get_bam_header<P>(file_path: P) -> Result<sam::Header>
660where
661    P: AsRef<Path>,
662{
663    let file_path = file_path.as_ref();
664
665    // Check that the file exists
666    if !file_path.exists() {
667        return Err(anyhow::Error::from(std::io::Error::new(
668            std::io::ErrorKind::NotFound,
669            format!("File not found: {}", file_path.display()),
670        )));
671    };
672
673    let mut reader = bam::io::indexed_reader::Builder::default()
674        .build_from_path(file_path)
675        .expect("Failed to open file");
676
677    let header = match reader.read_header() {
678        std::result::Result::Ok(header) => header,
679        Err(e) => {
680            debug!("Failed to read header using noodels falling back to samtools: {e}");
681
682            let header_samtools = std::process::Command::new("samtools")
683                .arg("view")
684                .arg("-H")
685                .arg(file_path)
686                .output()
687                .expect("Failed to run samtools")
688                .stdout;
689
690            let header_str =
691                String::from_utf8(header_samtools).expect("Failed to convert header to string");
692
693            // Slight hack here for CellRanger BAM files that are missing the version info
694            let header_string =
695                header_str.replace("@HD\tSO:coordinate\n", "@HD\tVN:1.6\tSO:coordinate\n");
696            let header_str = header_string.as_bytes();
697            let mut reader = sam::io::Reader::new(header_str);
698            reader
699                .read_header()
700                .expect("Failed to read header with samtools")
701        }
702    };
703    Ok(header)
704}
