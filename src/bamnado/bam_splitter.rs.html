1//! # BAM Splitter Module
2//!
3//! This module handles the splitting of BAM files into smaller subsets or filtering them
4//! based on user-defined criteria. It supports asynchronous processing and parallel execution
5//! to efficiently handle large genomic datasets.
6//!
7//! Key features include:
8//! *   Splitting BAM files by barcodes or other tags.
9//! *   Filtering reads during the splitting process.
10//! *   Asynchronous I/O for high performance.
11

29/// Filters BAM files based on read criteria.
30pub struct BamFilterer {
31    filepath: PathBuf,
32    output: PathBuf,
33    filter: BamReadFilter,
34}
35
36impl BamFilterer {
37    /// Creates a new `BamFilterer`.
38    ///
39    /// # Arguments
40    ///
41    /// * `filepath` - Path to the input BAM file.
42    /// * `output` - Path to the output BAM file.
43    /// * `filter` - Filter criteria for reads.
44    pub fn new(filepath: PathBuf, output: PathBuf, filter: BamReadFilter) -> Self {
45        BamFilterer {
46            filepath,
47            output,
48            filter,
49        }
50    }

52    /// Runs the filtering process asynchronously.
53    ///
54    /// This reads the input BAM, filters reads, and writes the result to the output file.
55    pub async fn split_async(&self) -> Result<()> {
56        let filepath = self.filepath.clone();
57        let outfile = self.output.clone();
58
59        let mut reader = File::open(&filepath).await.map(AsyncReader::new)?;
60        let header = reader.read_header().await;
61
62        let header = match header {
63            Ok(header) => header,
64            Err(_e) => get_bam_header(&filepath)?,
65        };
66
67        let index_path = self.filepath.with_extension("bam.bai");
68        let index_file = File::open(&index_path).await?;
69        let mut index_reader = bai::r#async::io::Reader::new(index_file);
70        let index = index_reader.read_index().await?;
71
72        // Make writer
73        let mut writer = AsyncWriter::new(File::create(outfile).await?);
74        writer.write_header(&header).await?;
75
76        // Get the chromosome sizes
77        let chromsizes = header
78            .reference_sequences()
79            .iter()
80            .map(|(name, seq)| (name.to_string(), seq.length().get() as u64))
81            .collect::<std::collections::HashMap<_, _>>();
82
83        let query_regions = chromsizes.iter().map(|(name, size)| {
84            let start = Position::try_from(1).unwrap();
85            let end = Position::try_from(*size as usize).unwrap();
86            Region::new(name.to_string(), start..=end)
87        });
88
89        let progress = ProgressBar::new(chromsizes.len() as u64);
90
91        for region in query_regions {
92            progress.inc(1);
93
94            // println!("Querying region: {:?}", region);
95
96            let mut query = reader.query(&header, &index, &region)?;
97            let mut record = bam::Record::default();
98
99            while query.read_record(&mut record).await? != 0 {
100                let is_valid = self.filter.is_valid(&record, Some(&header))?;
101                if is_valid {
102                    writer.write_record(&header, &record).await?;
103                }
104            }
105        }
106
107        progress.finish();
108        writer.shutdown().await?;
109
110        Ok(())
111    }

113    pub fn split(&self) -> Result<()> {
114        let filepath = self.filepath.clone();
115
116        // Determine the chunks over which to iterate
117        // Assuming here that the depth of the first file is the same as the rest
118        let bam_stats = BamStats::new(self.filepath.clone())?;
119        let genomic_chunks = bam_stats.genome_chunks(1e6 as u64)?;
120
121        // Set up required variables
122        let _chromsizes_refid = bam_stats
123            .chromsizes_ref_id()
124            .expect("Error getting chromsizes");
125
126        // Iterate over the genomic chunks and pileup the reads
127        let _n_total_chunks = genomic_chunks.len();
128
129        // Start a writing thread
130        let outfile = self.output.clone();
131
132        // Set-up for multithreaded writing
133        let (tx, rx) = unbounded();
134
135        // Start a writing thread
136        let handle = std::thread::spawn(move || {
137            let _reader = bam::io::indexed_reader::Builder::default()
138                .build_from_path(&filepath)
139                .expect("Error reading file");
140            let header = get_bam_header(&filepath).expect("Error reading header");
141
142            let mut writer = bam::io::writer::Builder {}
143                .build_from_path(outfile)
144                .expect("Error writing to file");
145
146            for chunk in rx.iter() {
147                for record in chunk {
148                    writer
149                        .write_record(&header, &record)
150                        .expect("Error writing record");
151                }
152            }
153        });
154
155        let filepath = self.filepath.clone();
156
157        // Process the genomic chunks filtering reads
158        genomic_chunks
159            .into_par_iter()
160            .for_each_with(tx, |tx, chunk| {
161                let mut reader = bam::io::indexed_reader::Builder::default()
162                    .build_from_path(&filepath)
163                    .expect("Error reading file");
164
165                let header = get_bam_header(&filepath).expect("Error reading header");
166
167                let records = reader
168                    .query(&header, &chunk)
169                    .expect("Error getting chunk of reads");
170
171                let filtered_records = records
172                    .records()
173                    .filter_map(|r| r.is_ok().then(|| r.unwrap()))
174                    .filter(|record| self.filter.is_valid(record, Some(&header)).unwrap_or(false))
175                    .collect::<Vec<_>>();
176
177                tx.send(filtered_records).expect("Error sending records");
178            });
179
180        handle.join().expect("Error joining threads");
181        Ok(())
182    }
