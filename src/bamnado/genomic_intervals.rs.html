21/// Returns the alignment span of a CIGAR operation.
22/// # Arguments
23/// * `cigar` - A reference to a CIGAR operation.
24/// # Returns
25/// The total length of the alignment span, which is the sum of all match, insertion,
26/// and deletion operations in the CIGAR string.
27fn alignment_span<C>(cigar: &C) -> Result<usize>
28where
29    C: Cigar,
30{
31    cigar.iter().try_fold(0, |acc, op| {
32        let op = op.context("Failed to parse CIGAR operation")?;
33        match op.kind() {
34            Kind::Match
35            | Kind::Insertion
36            | Kind::SequenceMatch
37            | Kind::Pad
38            | Kind::SequenceMismatch => Ok(acc + op.len()),
39            Kind::Deletion | Kind::Skip => Ok(acc),
40            _ => Ok(acc), // Ignore other kinds like SoftClip, HardClip, etc.
41        }
42    })
43}

45/// Represents coordinate shifts to be applied to reads.
46///
47/// The shifts can be applied to the 5' and 3' ends of forward and reverse reads.
48#[derive(Debug, Clone, Copy, PartialEq, Eq)]
49pub struct Shift {
50    /// Shift for the 5' end of forward reads.
51    pub five_prime: i32,
52    /// Shift for the 3' end of forward reads.
53    pub three_prime: i32,
54    /// Shift for the 5' end of reverse reads.
55    pub five_prime_reverse: i32,
56    /// Shift for the 3' end of reverse reads.
57    pub three_prime_reverse: i32,
58    index: usize,
59}
60
61impl Iterator for Shift {
62    type Item = i32;
63
64    fn next(&mut self) -> Option<Self::Item> {
65        match self.index {
66            0 => {
67                self.index += 1;
68                Some(self.five_prime)
69            }
70            1 => {
71                self.index += 1;
72                Some(self.three_prime)
73            }
74            2 => {
75                self.index += 1;
76                Some(self.five_prime_reverse)
77            }
78            3 => {
79                self.index += 1;
80                Some(self.three_prime_reverse)
81            }
82            _ => None,
83        }
84    }
85}

87impl FromStr for Shift {
88    type Err = anyhow::Error;
89
90    fn from_str(s: &str) -> Result<Self> {
91        let parts: Vec<&str> = s.split(',').collect();
92        if parts.len() != 4 {
93            return Err(anyhow!("Shift must be in the format '5,3,5r,3r'"));
94        }
95
96        let five_prime = parts[0].parse::<i32>().unwrap_or(0);
97        let three_prime = parts[1].parse::<i32>().unwrap_or(0);
98        let five_prime_reverse = parts[2].parse::<i32>().unwrap_or(0);
99        let three_prime_reverse = parts[3].parse::<i32>().unwrap_or(0);
100
101        Ok(Self {
102            five_prime,
103            three_prime,
104            five_prime_reverse,
105            three_prime_reverse,
106            index: 0,
107        })
108    }
109}

149/// Represents truncation values for reads.
150#[derive(Debug, Clone, Copy, PartialEq, Eq)]
151pub struct Truncate {
152    five_prime: Option<usize>,
153    three_prime: Option<usize>,
154}
155
156impl From<(Option<usize>, Option<usize>)> for Truncate {
157    fn from(truncate: (Option<usize>, Option<usize>)) -> Self {
158        Self {
159            five_prime: truncate.0,
160            three_prime: truncate.1,
161        }
162    }
163}
164
165impl FromStr for Truncate {
166    type Err = anyhow::Error;
167
168    fn from_str(s: &str) -> Result<Self> {
169        let parts: Vec<&str> = s.split(',').collect();
170        if parts.len() != 2 {
171            return Err(anyhow!("Truncate must be in the format '5,3'"));
172        }
173
174        let five_prime = parts[0].parse::<usize>().ok();
175        let three_prime = parts[1].parse::<usize>().ok();
176
177        Ok(Self {
178            five_prime,
179            three_prime,
180        })
181    }

184/// Helper struct for creating genomic intervals from BAM records.
185///
186/// It handles coordinate calculations, shifting, and truncation.
187pub struct IntervalMaker<'a> {
188    read: bam::Record,
189    header: &'a sam::Header,
190    chromsizes: &'a HashMap<usize, u64>,
191    filter: &'a BamReadFilter,
192    use_fragment: bool,
193    shift: Shift,
194    truncate: Option<Truncate>,
195}
196
197impl<'a> IntervalMaker<'a> {
198    /// Creates a new `IntervalMaker`.
199    pub fn new(
200        read: bam::Record,
201        header: &'a sam::Header,
202        chromsizes: &'a HashMap<usize, u64>,
203        filter: &'a BamReadFilter,
204        use_fragment: bool,
205        shift: Option<Shift>,
206        truncate: Option<Truncate>,
207    ) -> Self {
208        Self {
209            read,
210            header,
211            chromsizes,
212            filter,
213            use_fragment,
214            shift: shift.map_or_else(|| Shift::from([0, 0, 0, 0]), Shift::from),
215            truncate,
216        }
217    }
218
219    fn coords_fragment(&self) -> Option<(usize, usize, usize)> {
220        if !self.read.flags().is_first_segment() {
221            return None;
222        }
223
224        let template_length = self.read.template_length();
225
226        // If the template length is positive, we use the alignment start of the read.
227        // If it's negative, we use the mate alignment start.
228        // If it's zero, we return None as this has no valid genomic coordinates.
229        if template_length > 0 {
230            let start = self
231                .read
232                .alignment_start()
233                .context("Missing alignment start")
234                .ok()?
235                .ok()?
236                .get();
237            let end = start + template_length as usize;
238            Some((start, end, 0))
239        } else if template_length < 0 {
240            let start = self
241                .read
242                .mate_alignment_start()
243                .context("Missing mate alignment start")
244                .ok()?
245                .ok()?
246                .get();
247            let end = start + template_length.unsigned_abs() as usize;
248            Some((start, end, 0))
249        } else {
250            None
251        }
252    }
253
254    fn coords_read(&self) -> Option<(usize, usize, usize)> {
255        // For single reads, we use the alignment start and calculate the end based on the alignment span.
256        // If the read has no alignment start, we return None.
257        let start = self
258            .read
259            .alignment_start()
260            .context("Missing alignment start")
261            .ok()?
262            .ok()?
263            .get();
264        let alignment_span = alignment_span(&self.read.cigar())
265            .context("Failed to calculate alignment span")
266            .ok()?;
267        let end = start + alignment_span;
268        Some((start, end, 0))
269    }
270
271    fn coords_shifted(
272        &self,
273        mut start: usize,
274        mut end: usize,
275        _dtlen: usize,
276    ) -> Result<(usize, usize, i32)> {
277        // Changed return type to i32
278        let reverse = self.read.flags().is_reverse_complemented();
279        let first_in_template = self.read.flags().is_first_segment();
280
281        // Adjust start and end based on the shift values.
282        // The logic here depends on whether the read is reverse complemented and whether it is the first
283        // segment in the template.
284        // The shift values are applied differently based on these conditions.
285        // The shift array is expected to have the following meaning:
286        // [shift_5_prime, shift_3_prime, shift_5_prime_reverse, shift_3_prime_reverse]
287        let dtlen = match (reverse, first_in_template) {
288            (true, true) => {
289                // Reverse first read: apply shifts correctly for reverse strand
290                if self.shift.three_prime >= 0 {
291                    start = start.saturating_sub(self.shift.three_prime as usize);
292                } else {
293                    start = start.saturating_add((-self.shift.three_prime) as usize);
294                }
295                if self.shift.five_prime_reverse >= 0 {
296                    end = end.saturating_add(self.shift.five_prime_reverse as usize);
297                } else {
298                    end = end.saturating_sub((-self.shift.five_prime_reverse) as usize);
299                }
300                // For reverse reads, the template length change is the sum of the applied shifts
301                self.shift.five_prime_reverse + self.shift.three_prime
302            }
303            (true, false) => {
304                // Reverse second read: use reverse-specific shifts
305                if self.shift.three_prime_reverse >= 0 {
306                    start = start.saturating_sub(self.shift.three_prime_reverse as usize);
307                } else {
308                    start = start.saturating_add((-self.shift.three_prime_reverse) as usize);
309                }
310                if self.shift.five_prime_reverse >= 0 {
311                    end = end.saturating_add(self.shift.five_prime_reverse as usize);
312                } else {
313                    end = end.saturating_sub((-self.shift.five_prime_reverse) as usize);
314                }
315                self.shift.five_prime_reverse + self.shift.three_prime_reverse
316            }
317            (false, true) => {
318                // Forward first read: direct application of shifts
319                if self.shift.five_prime >= 0 {
320                    start = start.saturating_sub(self.shift.five_prime as usize);
321                } else {
322                    start = start.saturating_add((-self.shift.five_prime) as usize);
323                }
324                if self.shift.three_prime >= 0 {
325                    end = end.saturating_add(self.shift.three_prime as usize);
326                } else {
327                    end = end.saturating_sub((-self.shift.three_prime) as usize);
328                }
329                self.shift.five_prime + self.shift.three_prime
330            }
331            (false, false) => {
332                // Forward second read: mix of forward and reverse shifts
333                if self.shift.five_prime >= 0 {
334                    start = start.saturating_sub(self.shift.five_prime as usize);
335                } else {
336                    start = start.saturating_add((-self.shift.five_prime) as usize);
337                }
338                if self.shift.three_prime_reverse >= 0 {
339                    end = end.saturating_add(self.shift.three_prime_reverse as usize);
340                } else {
341                    end = end.saturating_sub((-self.shift.three_prime_reverse) as usize);
342                }
343                self.shift.five_prime + self.shift.three_prime_reverse
344            }
345        };
346
347        let ref_id = self
348            .read
349            .reference_sequence_id()
350            .context("Missing reference sequence ID")?
351            .context("Reference sequence ID is None")?;
352
353        let chromsize = self
354            .chromsizes
355            .get(&ref_id)
356            .ok_or_else(|| anyhow!("No chromosome size found for reference ID {ref_id}"))?;
357
358        // Clamp to valid genomic range
359        let start = start.max(1);
360        let end = min(*chromsize as usize, end);
361
362        Ok((start, end, dtlen))
363    }

365    /// Calculates the coordinates of the interval.
366    ///
367    /// Returns `(start, end, dtlen)` where `dtlen` is the change in template length.
368    pub fn coords(&self) -> Option<(usize, usize, i32)> {
369        // Changed return type
370        match self.filter.is_valid(&self.read, Some(self.header)) {
371            Ok(true) => {
372                let (start, end, dtlen) = if self.use_fragment {
373                    self.coords_fragment()?
374                } else {
375                    self.coords_read()?
376                };
377
378                let (start, end, dtlen) = if self.shift.into_iter().any(|x| x != 0) {
379                    self.coords_shifted(start, end, dtlen).ok()?
380                } else {
381                    (start, end, dtlen as i32)
382                };
383
384                let (start, end) = self.truncate(start, end);
385
386                Some((start, end, dtlen))
387            }
388            _ => None,
389        }
390    }
391
392    fn truncate(&self, start: usize, end: usize) -> (usize, usize) {
393        let mut start = start;
394        let mut end = end;
395
396        if self.truncate.is_none() {
397            return (start, end);
398        }
399
400        // If truncate is set, apply the truncation values.
401        // If truncate_5_prime is set, we adjust the start position.
402        // If truncate_3_prime is set, we adjust the end position.
403        // These values are optional, so we check if they are Some before applying them.
404        if let Some(truncate) = self.truncate {
405            if let Some(truncate_5_prime) = truncate.five_prime {
406                start = start.saturating_add(truncate_5_prime);
407            }
408
409            if let Some(truncate_3_prime) = truncate.three_prime {
410                end = end.saturating_sub(truncate_3_prime);
411            }
412        }
413
414        (start, end)
415    }

417    /// Creates a new BAM record with modified coordinates.
418    pub fn record(&self) -> Option<RecordBuf> {
419        let (start, _end, dtlen) = self.coords()?;
420
421        let mut new_record = RecordBuf::try_from_alignment_record(self.header, &self.read).ok()?;
422
423        *new_record.alignment_start_mut() = Position::new(start);
424
425        // Update template length based on the delta from coordinate modifications
426        let original_template_length = self.read.template_length();
427
428        // Apply the delta to the template length, preserving sign
429        let new_template_length = if original_template_length >= 0 {
430            original_template_length.saturating_add(dtlen)
431        } else {
432            original_template_length.saturating_sub(dtlen)
433        };
434
435        *new_record.template_length_mut() = new_template_length;
436
437        Some(new_record)
438    }
