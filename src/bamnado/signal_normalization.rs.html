1//! # Signal Normalization Module
2//!
3//! This module provides methods for normalizing genomic coverage signals.
4//! It supports common normalization techniques used in sequencing analysis, such as:
5//! *   **Raw**: No normalization.
6//! *   **RPKM**: Reads Per Kilobase per Million mapped reads.
7//! *   **CPM**: Counts Per Million mapped reads.
8
9use anyhow::Result;
10use log::warn;
11
12/// Normalization methods for genomic coverage signals.
13#[derive(Debug, Clone, clap::ValueEnum)]
14pub enum NormalizationMethod {
15    /// No normalization.
16    Raw,
17    /// Reads Per Kilobase per Million mapped reads.
18    RPKM,
19    /// Counts Per Million mapped reads.
20    CPM,
21}
22
23impl NormalizationMethod {
24    /// Parse a string into a `NormalizationMethod`.
25    ///
26    /// # Arguments
27    ///
28    /// * `s` - The string to parse.
29    #[allow(clippy::should_implement_trait)]
30    pub fn from_str(s: &str) -> Result<NormalizationMethod> {
31        match s.to_lowercase().as_str() {
32            "raw" => Ok(NormalizationMethod::Raw),
33            "rpkm" => Ok(NormalizationMethod::RPKM),
34            "cpm" => Ok(NormalizationMethod::CPM),
35            _ => {
36                warn!("Unknown normalization method: {s}. Defaulting to Raw");
37                Ok(NormalizationMethod::Raw)
38            }
39        }
40    }
41
42    /// Calculate the scaling factor for the given normalization method.
43    /// # Arguments:
44    /// * `base_scale` - The base scaling factor (e.g., 1.0 for raw counts).
45    /// * `bin_size` - The size of the genomic bin in base pairs.
46    /// * `n_reads` - The total number of reads in the dataset.
47    /// # Returns:
48    /// A scaling factor to apply to raw counts based on the normalization method.
49    /// This factor is used to adjust the raw counts to account for differences in sequencing depth and
50    /// genomic feature length.
51    pub fn scale_factor(&self, base_scale: f32, bin_size: u64, n_reads: u64) -> f64 {
52        let base = base_scale as f64;
53
54        match self {
55            Self::Raw => base,
56
57            Self::CPM => {
58                // Counts Per Million: normalize by library size
59                base * (1_000_000.0 / n_reads as f64)
60            }
61
62            Self::RPKM => {
63                // Reads Per Kilobase per Million: normalize by both length and library size
64                let reads_per_million = 1_000_000.0 / n_reads as f64;
65                let per_kilobase = 1_000.0 / bin_size as f64; // Convert bp to kb
66                base * reads_per_million * per_kilobase
67            }
68        }
69    }
70}

72#[cfg(test)]
73mod tests {
74    use super::*;
75
76    #[test]
77    fn test_scaling_factors() {
78        let bin_size = 1000; // 1kb bins
79        let n_reads = 10_000_000; // 10M reads
80        let base_scale = 1.0;
81
82        let raw_factor = NormalizationMethod::Raw.scale_factor(base_scale, bin_size, n_reads);
83        let cpm_factor = NormalizationMethod::CPM.scale_factor(base_scale, bin_size, n_reads);
84        let rpkm_factor = NormalizationMethod::RPKM.scale_factor(base_scale, bin_size, n_reads);
85
86        println!("Raw factor: {raw_factor}"); // Should be 1.0
87        println!("CPM factor: {cpm_factor}"); // Should be 0.1 (1M/10M)
88        println!("RPKM factor: {rpkm_factor}"); // Should be 0.1 (1M/10M * 1000/1K)
89
90        // For 1000 reads in a 1kb bin with 10M total reads:
91        let raw_value = 1000.0 * raw_factor; // 1000
92        let cpm_value = 1000.0 * cpm_factor; // 100 CPM
93        let rpkm_value = 1000.0 * rpkm_factor; // 100 RPKM
94
95        assert_eq!(raw_value, 1000.0);
96        assert_eq!(cpm_value, 100.0);
97        assert_eq!(rpkm_value, 100.0); // Changed from 0.1 to 100.0
98    }
