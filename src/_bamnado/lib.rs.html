1//! # BamNado Python Interface
2//!
3//! This crate provides the Python bindings for the `bamnado` library using `pyo3`.
4//! It exposes high-performance Rust implementations of BAM processing tools to Python.
5//!
6//! ## Exposed Functions
7//!
8//! *   `get_signal_for_chromosome`: Calculates coverage signal for a chromosome.
9
10use bamnado::{bam_utils, coverage_analysis, read_filter::BamReadFilter, signal_normalization};
11use ndarray::prelude::*;
12use numpy::{PyArray1, prelude::*};
13use pyo3::exceptions::{PyKeyError, PyRuntimeError};
14use pyo3::prelude::*;
15
16fn anyhow_to_pyerr(err: anyhow::Error) -> PyErr {
17    PyRuntimeError::new_err(err.to_string())
18}

20#[pymodule]

24    /// Get the signal for a specific chromosome from a BAM file.
25    ///
26    /// This function calculates the coverage signal for a given chromosome, optionally scaling it
27    /// and using fragment length information.
28    ///
29    /// Args:
30    ///     bam_path (str): Path to the BAM file.
31    ///     chromosome_name (str): Name of the chromosome to analyze.
32    ///     bin_size (int): Size of the bins for coverage calculation.
33    ///     scale_factor (float): Factor to scale the signal by.
34    ///     use_fragment (bool): Whether to use fragment length for coverage (True) or just read start (False).
35    ///     ignore_scaffold_chromosomes (bool): Whether to ignore scaffold chromosomes during analysis.
36    ///
37    /// Returns:
38    ///     numpy.ndarray: A 1D numpy array of floats representing the signal across the chromosome.
39    ///
40    /// Raises:
41    ///     RuntimeError: If there is an error reading the BAM file or processing the data.
42    ///     KeyError: If the specified chromosome is not found in the BAM file.
43    #[pyfunction]
44    #[pyo3(
45        text_signature = "(bam_path, chromosome_name, bin_size, scale_factor, use_fragment, ignore_scaffold_chromosomes)"
46    )]
47    fn get_signal_for_chromosome(
48        py: Python,
49        bam_path: &str,
50        chromosome_name: &str,
51        bin_size: u64,
52        scale_factor: f32,
53        use_fragment: bool,
54        ignore_scaffold_chromosomes: bool,
55    ) -> PyResult<Py<PyArray1<f32>>> {
56        let stats = bam_utils::BamStats::new(bam_path.into()).map_err(anyhow_to_pyerr)?;
57        let chrom_size = match stats
58            .chromsizes_ref_name()
59            .map_err(anyhow_to_pyerr)?
60            .get(chromosome_name)
61        {
62            Some(size) => *size,
63            None => {
64                return Err(PyKeyError::new_err(format!(
65                    "Chromosome {} not found in BAM file.",
66                    chromosome_name
67                )));
68            }
69        };
70
71        let bam_pileup = coverage_analysis::BamPileup::new(
72            bam_path.into(),
73            bin_size,
74            signal_normalization::NormalizationMethod::Raw,
75            scale_factor,
76            use_fragment,
77            BamReadFilter::default(),
78            true,
79            ignore_scaffold_chromosomes,
80            None,
81            None,
82        );
83
84        let signal = bam_pileup
85            .pileup_chromosome(chromosome_name)
86            .map_err(anyhow_to_pyerr)?;
87
88        // Put the signal into an ndarray then convert to numpy array
89        let mut array = Array1::zeros(chrom_size as usize);
90        for interval in signal {
91            let start = interval.start;
92            let end = interval.stop;
93            let value = interval.val as f32;
94            array.slice_mut(s![start..end]).fill(value);
95        }
96
97        let numpy_array = array.into_pyarray(py).unbind();
98        Ok(numpy_array)
99    }
100
101    #[pyfunction]
102    fn __version__() -> &'static str {
103        env!("CARGO_PKG_VERSION")
104    }
105}
